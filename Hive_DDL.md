### Create Database:
    CREATE DATABASE IF NOT EXISTS [database_name];
    
### Show Database:
    SHOW DATABASES;

### Create Tables in Hive:
    CREATE TABLE IF NOT EXISTS [table_name](
        variable1 datatype1,
        variable2 mdatatype2)
        ROW FORMAT DELIMITED
        FIELDS TERMINATED BY ‘,’
        STORED AS TEXTFILE
